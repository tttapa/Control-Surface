25Teensy2_USBDeviceMIDIBackend::read() {
26 // https://github.com/PaulStoffregen/cores/blob/73ea157600a7082686d9cc48786a73caa7567da9/usb_midi/usb_api.cpp#L195
27 uint8_t c, intr_state;
28
29 intr_state = SREG;
30 cli();
31 if (!usb_configuration) {
32 SREG = intr_state;
33 return MIDIUSBPacket_t {0};
34 }
35 UENUM = MIDI_RX_ENDPOINT;
36retry:
37 c = UEINTX;
38 if (!(c & (1 << RWAL))) {

42 }
43 SREG = intr_state;
44 return MIDIUSBPacket_t {0};
45 }

49 packet.data[2] = UEDATX;
50 packet.data[3] = UEDATX;
51 if (!(UEINTX & (1 << RWAL)))
52 UEINTX = 0x6B;
53 SREG = intr_state;

58inline void Teensy2_USBDeviceMIDIBackend::write(MIDIUSBPacket_t data) {
59 uint8_t intr_state, timeout;
60
61 if (!usb_configuration)
62 return;
63 intr_state = SREG;
64 cli();
65 UENUM = MIDI_TX_ENDPOINT;
66 timeout = UDFNUML + 2;
67 while (1) {
68 // are we ready to transmit?
69 if (UEINTX & (1 << RWAL))
70 break;
71 SREG = intr_state;
72 if (UDFNUML == timeout)
73 return;
74 if (!usb_configuration)
75 return;
76 intr_state = SREG;
77 cli();
78 UENUM = MIDI_TX_ENDPOINT;
79 }
80 UEDATX = data.data[0];
81 UEDATX = data.data[1];
82 UEDATX = data.data[2];
83 UEDATX = data.data[3];
84 if (!(UEINTX & (1 << RWAL)))
85 UEINTX = 0x3A;
86 SREG = intr_state;
87}
