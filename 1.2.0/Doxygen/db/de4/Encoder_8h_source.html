61 // All the data needed by interrupts is consolidated into this ugly struct
62 // to facilitate assembly language optimizing of the speed critical update.
63 // The assembly code uses auto-incrementing addressing modes, so the struct
64 // must remain in exactly this order.
65 typedef struct {
66  volatile IO_REG_TYPE *pin1_register;
67  volatile IO_REG_TYPE *pin2_register;
68  IO_REG_TYPE pin1_bitmask;
69  IO_REG_TYPE pin2_bitmask;
70  uint8_t state;
71  int32_t position;
72 } Encoder_internal_state_t;

88  Encoder(uint8_t pin1, uint8_t pin2)
89  : pin1(pin1), pin2(pin2)
90  {
91  // It's much faster to use the GPIO registers directly, rather than
92  // calling digitalRead every time we need to read a pin.
93  // digitalRead looks up the register and bitmasks every time it's called
94  // but here, we look them up once in the constructor.
95  encoder.pin1_register = PIN_TO_BASEREG(pin1);
96  encoder.pin1_bitmask = PIN_TO_BITMASK(pin1);
97  encoder.pin2_register = PIN_TO_BASEREG(pin2);
98  encoder.pin2_bitmask = PIN_TO_BITMASK(pin2);
99  encoder.position = 0;
100  }

111  *this = enc_util::move(other);

114  Encoder &operator=(Encoder &&other) {
115 #ifdef ENCODER_USE_INTERRUPTS
116  // First swap the normal member variables:
117  enc_util::swap(this->pin1, other.pin1);
118  enc_util::swap(this->pin2, other.pin2);
119  enc_util::swap(this->interrupts_in_use, other.interrupts_in_use);
120  // Next, update the pointers in interruptArgs:
121  // When interrupts are in use, there is a global interrupt context
122  // variable that holds a pointer to the encoders that are being swapped
123  // or moved.
124  // After moving, this pointer would no longer be valid, so it has to be
125  // changed to point to the new encoder object.
126  // Calling attachInterrupt is not necessary, because this should already
127  // have happened in the begin method if interrupts_in_use is nonzero.
128  // Before messing with the state variables that can be changed or
129  // accessed from within an ISR, we have to disable interrupts.
130  noInterrupts();
131  enc_util::swap(this->encoder, other.encoder);
132  if (this->interrupts_in_use > 0) {
133  auto int1 = digitalPinToInterrupt(this->pin1);
134  if (int1 != NOT_AN_INTERRUPT)
135  interruptArgs[int1] = &this->encoder;
136  auto int2 = digitalPinToInterrupt(this->pin2);
137  if (int2 != NOT_AN_INTERRUPT)
138  interruptArgs[int2] = &this->encoder;
139  }
140  if (other.interrupts_in_use > 0) {
141  auto int1 = digitalPinToInterrupt(other.pin1);
142  if (int1 != NOT_AN_INTERRUPT)
143  interruptArgs[int1] = &other.encoder;
144  auto int2 = digitalPinToInterrupt(other.pin2);
145  if (int2 != NOT_AN_INTERRUPT)
146  interruptArgs[int2] = &other.encoder;
147  }
148  interrupts();
149 #else
150  // If interrupts are disabled, we don't have to worry about dangling
151  // pointer in interrupt contexts, we can just swap the encoder state
152  // member variables.
153  enc_util::swap(this->encoder, other.encoder);
154 #endif
155  return *this;
156  }

170  void begin() {
171 #ifdef INPUT_PULLUP
172  pinMode(pin1, INPUT_PULLUP);
173  pinMode(pin2, INPUT_PULLUP);
174 #else
175  pinMode(pin1, INPUT);
176  digitalWrite(pin1, HIGH);
177  pinMode(pin2, INPUT);
178  digitalWrite(pin2, HIGH);
179 #endif
180  // allow time for a passive R-C filter to charge
181  // through the pullup resistors, before reading
182  // the initial state
183  delayMicroseconds(2000);
184  uint8_t s = 0;
185  if (DIRECT_PIN_READ(encoder.pin1_register, encoder.pin1_bitmask))
186  s |= 1;
187  if (DIRECT_PIN_READ(encoder.pin2_register, encoder.pin2_bitmask))
188  s |= 2;
189  encoder.state = s;
190 
191 #ifdef ENCODER_USE_INTERRUPTS
192  attachInterruptCtx(digitalPinToInterrupt(pin1));
193  attachInterruptCtx(digitalPinToInterrupt(pin2));
194 #endif
195  }
196 
198  void end() {

202  detachInterruptCtx(digitalPinToInterrupt(pin2));
203  }
204 #endif
205  }

208  void attachInterruptCtx(int interrupt) {
209  if (interrupt != NOT_AN_INTERRUPT) {
210  interruptArgs[interrupt] = &encoder;
211  ++interrupts_in_use;
212  attachInterrupt(interrupt, EncoderISRs::getISR(interrupt), CHANGE);
213  }

215  void detachInterruptCtx(int interrupt) {
216  if (interrupt != NOT_AN_INTERRUPT) {
217  detachInterrupt(interrupt);
218  --interrupts_in_use;
219  interruptArgs[interrupt] = nullptr;

324  // update() is not meant to be called from outside Encoder,
325  // but it is public to allow static interrupt routines.
326  // DO NOT call update() directly from sketches.
327  static void update(Encoder_internal_state_t *arg) {
328 #if defined(__AVR__)
329  // The compiler believes this is just 1 line of code, so
330  // it will inline this function into each interrupt
331  // handler. That's a tiny bit faster, but grows the code.
332  // Especially when used with ENCODER_OPTIMIZE_INTERRUPTS,
333  // the inline nature allows the ISR prologue and epilogue
334  // to only save/restore necessary registers, for very nice
335  // speed increase.
336  asm volatile (
337  "ld r30, X+" "\n\t"
338  "ld r31, X+" "\n\t"
339  "ld r24, Z" "\n\t" // r24 = pin1 input
340  "ld r30, X+" "\n\t"
341  "ld r31, X+" "\n\t"
342  "ld r25, Z" "\n\t" // r25 = pin2 input
343  "ld r30, X+" "\n\t" // r30 = pin1 mask
344  "ld r31, X+" "\n\t" // r31 = pin2 mask
345  "ld r22, X" "\n\t" // r22 = state
346  "andi r22, 3" "\n\t"
347  "and r24, r30" "\n\t"
348  "breq L%=1" "\n\t" // if (pin1)
349  "ori r22, 4" "\n\t" // state |= 4
350  "L%=1:" "and r25, r31" "\n\t"
351  "breq L%=2" "\n\t" // if (pin2)
352  "ori r22, 8" "\n\t" // state |= 8
353  "L%=2:" "ldi r30, lo8(pm(L%=table))" "\n\t"
354  "ldi r31, hi8(pm(L%=table))" "\n\t"
355  "add r30, r22" "\n\t"
356  "adc r31, __zero_reg__" "\n\t"
357  "asr r22" "\n\t"
358  "asr r22" "\n\t"
359  "st X+, r22" "\n\t" // store new state
360  "ld r22, X+" "\n\t"
361  "ld r23, X+" "\n\t"
362  "ld r24, X+" "\n\t"
363  "ld r25, X+" "\n\t"
364  "ijmp" "\n\t" // jumps to update_finishup()
365  // TODO move this table to another static function,
366  // so it doesn't get needlessly duplicated. Easier
367  // said than done, due to linker issues and inlining
368  "L%=table:" "\n\t"
369  "rjmp L%=end" "\n\t" // 0
370  "rjmp L%=plus1" "\n\t" // 1
371  "rjmp L%=minus1" "\n\t" // 2
372  "rjmp L%=plus2" "\n\t" // 3
373  "rjmp L%=minus1" "\n\t" // 4
374  "rjmp L%=end" "\n\t" // 5
375  "rjmp L%=minus2" "\n\t" // 6
376  "rjmp L%=plus1" "\n\t" // 7
377  "rjmp L%=plus1" "\n\t" // 8
378  "rjmp L%=minus2" "\n\t" // 9
379  "rjmp L%=end" "\n\t" // 10
380  "rjmp L%=minus1" "\n\t" // 11
381  "rjmp L%=plus2" "\n\t" // 12
382  "rjmp L%=minus1" "\n\t" // 13
383  "rjmp L%=plus1" "\n\t" // 14
384  "rjmp L%=end" "\n\t" // 15
385  "L%=minus2:" "\n\t"
386  "subi r22, 2" "\n\t"
387  "sbci r23, 0" "\n\t"
388  "sbci r24, 0" "\n\t"
389  "sbci r25, 0" "\n\t"
390  "rjmp L%=store" "\n\t"
391  "L%=minus1:" "\n\t"
392  "subi r22, 1" "\n\t"
393  "sbci r23, 0" "\n\t"
394  "sbci r24, 0" "\n\t"
395  "sbci r25, 0" "\n\t"
396  "rjmp L%=store" "\n\t"
397  "L%=plus2:" "\n\t"
398  "subi r22, 254" "\n\t"
399  "rjmp L%=z" "\n\t"
400  "L%=plus1:" "\n\t"
401  "subi r22, 255" "\n\t"
402  "L%=z:" "sbci r23, 255" "\n\t"
403  "sbci r24, 255" "\n\t"
404  "sbci r25, 255" "\n\t"
405  "L%=store:" "\n\t"
406  "st -X, r25" "\n\t"
407  "st -X, r24" "\n\t"
408  "st -X, r23" "\n\t"
409  "st -X, r22" "\n\t"
410  "L%=end:" "\n"
411  : : "x" (arg) : "r22", "r23", "r24", "r25", "r30", "r31");
412 #else
413  uint8_t p1val = DIRECT_PIN_READ(arg->pin1_register, arg->pin1_bitmask);
414  uint8_t p2val = DIRECT_PIN_READ(arg->pin2_register, arg->pin2_bitmask);
415  uint8_t state = arg->state & 3;
416  if (p1val) state |= 4;
417  if (p2val) state |= 8;
418  arg->state = (state >> 2);
419  switch (state) {
420  case 1: case 7: case 8: case 14:
421  arg->position++;
422  return;
423  case 2: case 4: case 11: case 13:
424  arg->position--;
425  return;
426  case 3: case 12:
427  arg->position += 2;
428  return;
429  case 6: case 9:

431  return;
432  }
433 #endif
