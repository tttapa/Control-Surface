104template <class BytePuller>
105inline MIDIReadEvent USBMIDI_Parser::pull(BytePuller &&puller) {
106 // First try resuming the parser, we might have a stored packet that has to
107 // be parsed first.
108 MIDIReadEvent evt = resume();
109 if (evt != MIDIReadEvent::NO_MESSAGE)
110 return evt;
111
112 // If resumption didn't produce a message, read new packets from the input
113 // and parse them until either we get a message, or until the input runs out
114 // of new packets.
115 MIDIUSBPacket_t midiPacket;
116 while (puller.pull(midiPacket)) {
117 evt = feed(midiPacket);
118 if (evt != MIDIReadEvent::NO_MESSAGE)
119 return evt;
120 }
121 return MIDIReadEvent::NO_MESSAGE;
122}
