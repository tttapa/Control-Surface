92 MIDIReadEvent read() {
93 if (!midi.read()) // Update the MIDI input and check if there's
94 return MIDIReadEvent::NO_MESSAGE; // a new message available
95 auto type = midi.getType();
96 if (type <= uint8_t(MIDIMessageType::PitchBend)) { // Channel
97 parser.updateChannelMessage(midi);
98 return MIDIReadEvent::CHANNEL_MESSAGE;
99 } else if (type == uint8_t(MIDIMessageType::SysExStart)) { // SysEx
100 parser.updateSysExMessage(midi);
101 return MIDIReadEvent::SYSEX_MESSAGE;
102 } else if (type <= uint8_t(MIDIMessageType::TuneRequest)) { // SysComm
103 parser.updateSysCommonMessage(midi);
104 return MIDIReadEvent::SYSCOMMON_MESSAGE;
105 } else if (type == uint8_t(MIDIMessageType::SysExEnd)) { // SysEx
106 // ignore
107 } else { // Real-Time
108 parser.updateRealTimeMessage(midi);
109 return MIDIReadEvent::REALTIME_MESSAGE;
110 }
111 return MIDIReadEvent::NO_MESSAGE;
112 }
