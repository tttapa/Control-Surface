50 uint8_t getStartOn() const {
51 int8_t position = getPosition();
52 if (position == 0)
53 return 0;
54 int8_t value = position - 1;
55 switch (getMode()) {
56 case SingleDot: return value;
57 case BoostCut: return std::min(+value, 5);
58 case Wrap: return 0;
59 case Spread: return std::max(5 - value, 0);
60 default: return 0;
61 }

68 case SingleDot: return value;
69 case BoostCut: return std::max(+value, 6);
70 case Wrap: return value;
71 case Spread: return std::min(5 + value, 11);
72 default: return 0;
73 }
