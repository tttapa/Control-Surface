48void StreamDebugMIDI_Base::sendChannelMessageImpl(Stream &stream,
49 ChannelMessage msg) {
50 uint8_t messageType = (msg.header >> 4) - 8;
51 if (messageType >= 7)
52 return;
53
54 DEBUG_LOCK_MUTEX
55 if (prefix != nullptr)
56 stream << prefix << ' ';
57 if (msg.hasTwoDataBytes())
58 stream << DebugMIDIMessageNames::MIDIStatusTypeNames[messageType]
59 << F("Channel: ") << msg.getChannel().getOneBased()
60 << F("\tData 1: 0x") << hex << msg.getData1()
61 << F("\tData 2: 0x") << msg.getData2() << dec;
62 else
63 stream << DebugMIDIMessageNames::MIDIStatusTypeNames[messageType]
64 << F("Channel: ") << msg.getChannel().getOneBased()
65 << F("\tData 1: 0x") << hex << msg.getData1() << dec;
66 if (msg.getMessageType() == msg.PitchBend)
67 stream << " (" << msg.getData14bit() << ')';
68 if (msg.getCable() != Cable_1)
69 stream << F("\tCable: ") << msg.getCable().getOneBased();
70 stream << "\r\n";
71}
