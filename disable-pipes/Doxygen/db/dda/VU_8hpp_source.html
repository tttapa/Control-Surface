47 Changed update(uint8_t data) {
48 switch (data) {
49 case 0xF: { // clear overload
50 Changed changed = overload ? OverloadChanged : NothingChanged;
51 overload = false;
52 return changed;

54 case 0xE: { // set overload
55 Changed changed = !overload ? OverloadChanged : NothingChanged;
