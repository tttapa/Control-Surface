187 spi.transfer(0x00); // No-Op

189 }
190 spi.transfer(opcode);
191 spi.transfer(value);
192 for (c++; c < chainlength; c++) {
193 spi.transfer(0x00); // No-Op
194 spi.transfer(0x00);
195 }
196 ExtIO::digitalWrite(loadPin, HIGH);
